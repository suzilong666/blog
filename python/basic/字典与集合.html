字典与集合 ​

官方文档

字典和集合是 Python 中重要的数据结构。字典用于存储键值对，提供快速查找；集合用于存储唯一元素，支持集合运算。本文将详细介绍两者的用法和应用场景。

字典（Dictionary） ​

创建字典 ​

python
# 空字典
empty = {}
empty = dict()

# 字面量创建
person = {"name": "张三", "age": 25, "city": "北京"}

# dict() 构造函数
person = dict(name="张三", age=25, city="北京")

# 从键值对列表创建
pairs = [("name", "李四"), ("age", 30)]
person = dict(pairs)
print(person)  # {'name': '李四', 'age': 30}

# 从两个列表创建
keys = ["name", "age"]
values = ["王五", 28]
person = dict(zip(keys, values))
print(person)  # {'name': '王五', 'age': 28}

访问元素 ​

python
person = {"name": "张三", "age": 25, "city": "北京"}

# 通过键访问
print(person["name"])    # 张三
print(person.get("age")) # 25

# get() 方法（安全访问）
print(person.get("phone"))          # None（键不存在）
print(person.get("phone", "未知"))  # 未知（提供默认值）

# 直接访问不存在的键会报错
# print(person["phone"])  # KeyError!

修改字典 ​

python
person = {"name": "张三", "age": 25}

# 添加/修改键值对
person["phone"] = "13800138000"  # 添加
person["age"] = 26                # 修改
print(person)
# {'name': '张三', 'age': 26, 'phone': '13800138000'}

# 更新多个键值对
person.update({"city": "上海", "age": 27})
print(person)
# {'name': '张三', 'age': 27, 'phone': '13800138000', 'city': '上海'}

# 删除键值对
del person["phone"]
print(person)
# {'name': '张三', 'age': 27, 'city': '上海'}

removed = person.pop("city")
print(removed)  # 上海

# 弹出最后一项（Python 3.7+）
last = person.popitem()
print(last)  # ('age', 27)

遍历字典 ​

python
person = {"name": "张三", "age": 25, "city": "北京"}

# 遍历键
for key in person:
    print(key)
# name
# age
# city

# 遍历键（显式）
for key in person.keys():
    print(key)

# 遍历值
for value in person.values():
    print(value)
# 张三
# 25
# 北京

# 遍历键值对
for key, value in person.items():
    print(f"{key}: {value}")
# name: 张三
# age: 25
# city: 北京

字典推导式 ​

python
# 基本推导式
squares = {x: x ** 2 for x in range(1, 6)}
print(squares)
# {1: 1, 2: 4, 3: 9, 4: 16, 5: 25}

# 带条件的推导式
even_squares = {x: x ** 2 for x in range(10) if x % 2 == 0}
print(even_squares)
# {0: 0, 2: 4, 4: 16, 6: 36, 8: 64}

# 交换键值
original = {"a": 1, "b": 2, "c": 3}
swapped = {v: k for k, v in original.items()}
print(swapped)  # {1: 'a', 2: 'b', 3: 'c'}

常用方法 ​

python
person = {"name": "张三", "age": 25}

# 获取所有键
print(list(person.keys()))    # ['name', 'age']

# 获取所有值
print(list(person.values()))  # ['张三', 25]

# 获取所有键值对
print(list(person.items()))   # [('name', '张三'), ('age', 25)]

# 检查键是否存在
print("name" in person)       # True
print("phone" not in person)  # True

# 字典长度
print(len(person))            # 2

# 清空字典
person.clear()
print(person)  # {}

嵌套字典 ​

python
students = {
    "001": {"name": "张三", "score": 90},
    "002": {"name": "李四", "score": 85},
    "003": {"name": "王五", "score": 92}
}

# 访问嵌套数据
print(students["001"]["name"])  # 张三

# 遍历
for student_id, info in students.items():
    print(f"{student_id}: {info['name']} - {info['score']}分")

集合（Set） ​

创建集合 ​

python
# 空集合（不能用 {}，那是空字典）
empty = set()

# 字面量创建
fruits = {"apple", "banana", "orange"}

# 从列表创建
numbers = set([1, 2, 3, 2, 1])
print(numbers)  # {1, 2, 3}（自动去重）

# 从字符串创建
chars = set("hello")
print(chars)  # {'h', 'e', 'l', 'o'}

基本操作 ​

python
fruits = {"apple", "banana", "orange"}

# 添加元素
fruits.add("grape")
print(fruits)  # {'apple', 'banana', 'orange', 'grape'}

# 添加多个元素
fruits.update(["melon", "pear"])
print(fruits)

# 删除元素
fruits.remove("banana")  # 不存在会报错
fruits.discard("grape")  # 不存在不会报错
removed = fruits.pop()   # 随机删除并返回
print(removed)

# 清空集合
fruits.clear()
print(fruits)  # set()

集合运算 ​

python
set_a = {1, 2, 3, 4, 5}
set_b = {4, 5, 6, 7, 8}

# 并集
print(set_a | set_b)           # {1, 2, 3, 4, 5, 6, 7, 8}
print(set_a.union(set_b))      # 同上

# 交集
print(set_a & set_b)           # {4, 5}
print(set_a.intersection(set_b))  # 同上

# 差集
print(set_a - set_b)           # {1, 2, 3}
print(set_a.difference(set_b)) # 同上

# 对称差集（只在其中一个集合的元素）
print(set_a ^ set_b)           # {1, 2, 3, 6, 7, 8}
print(set_a.symmetric_difference(set_b))  # 同上

集合关系 ​

python
set_a = {1, 2, 3}
set_b = {1, 2, 3, 4, 5}

# 子集
print(set_a <= set_b)              # True
print(set_a.issubset(set_b))       # True

# 超集
print(set_b >= set_a)              # True
print(set_b.issuperset(set_a))     # True

# 不相交
set_c = {6, 7}
print(set_a.isdisjoint(set_c))     # True

集合推导式 ​

python
# 基本推导式
squares = {x ** 2 for x in range(1, 6)}
print(squares)  # {1, 4, 9, 16, 25}

# 带条件的推导式
evens = {x for x in range(10) if x % 2 == 0}
print(evens)  # {0, 2, 4, 6, 8}

字典 vs 集合 ​

特性字典集合
语法{key: value}{element}
存储内容键值对唯一元素
主要用途快速查找去重、集合运算
有序性Python 3.7+ 保持插入顺序无序

常见错误 ​

1. 使用不可哈希类型作为键 ​

python
# 列表不能作为字典键
# my_dict = {[1, 2]: "value"}  # TypeError!

# 使用元组代替
my_dict = {(1, 2): "value"}  # 正确

2. 混淆空字典和空集合 ​

python
empty_dict = {}      # 这是空字典
empty_set = set()    # 这才是空集合

print(type({}))      # <class 'dict'>
print(type(set()))   # <class 'set'>

3. 遍历时修改字典 ​

python
my_dict = {"a": 1, "b": 2, "c": 3}

# 危险！不要在遍历时修改
# for key in my_dict:
#     if key == "b":
#         del my_dict[key]  # RuntimeError!

# 正确做法：先收集要删除的键
keys_to_delete = [key for key in my_dict if my_dict[key] == 2]
for key in keys_to_delete:
    del my_dict[key]

实际应用示例 ​

1. 词频统计 ​

python
def word_frequency(text):
    words = text.lower().split()
    freq = {}
    for word in words:
        freq[word] = freq.get(word, 0) + 1
    return freq

text = "hello world hello python world hello"
print(word_frequency(text))
# {'hello': 3, 'world': 2, 'python': 1}

2. 去重 ​

python
# 列表去重
numbers = [1, 2, 3, 2, 1, 4, 5, 4]
unique = list(set(numbers))
print(unique)  # [1, 2, 3, 4, 5]（顺序可能改变）

# 保持顺序的去重
def unique_ordered(items):
    seen = set()
    result = []
    for item in items:
        if item not in seen:
            seen.add(item)
            result.append(item)
    return result

print(unique_ordered(numbers))  # [1, 2, 3, 4, 5]

3. 缓存实现 ​

python
def fibonacci(n, cache={}):
    """带缓存的斐波那契数列"""
    if n in cache:
        return cache[n]
    if n < 2:
        return n
    result = fibonacci(n - 1) + fibonacci(n - 2)
    cache[n] = result
    return result

print(fibonacci(50))  # 快速计算

4. 分组数据 ​

python
students = [
    {"name": "张三", "grade": "A"},
    {"name": "李四", "grade": "B"},
    {"name": "王五", "grade": "A"},
    {"name": "赵六", "grade": "C"}
]

# 按成绩分组
groups = {}
for student in students:
    grade = student["grade"]
    if grade not in groups:
        groups[grade] = []
    groups[grade].append(student["name"])

print(groups)
# {'A': ['张三', '王五'], 'B': ['李四'], 'C': ['赵六']}

最佳实践 ​

  1. 使用 get() 安全访问：避免 KeyError
  2. 利用 defaultdict：简化默认值处理
  3. 集合用于去重：比列表更高效
  4. 字典推导式：简洁创建字典
  5. 注意可变性问题：不要用可变对象作为键

defaultdict 示例 ​

python
from collections import defaultdict

# 普通字典
freq = {}
for word in ["apple", "banana", "apple"]:
    freq[word] = freq.get(word, 0) + 1

# defaultdict（更简洁）
freq = defaultdict(int)
for word in ["apple", "banana", "apple"]:
    freq[word] += 1

print(dict(freq))  # {'apple': 2, 'banana': 1}

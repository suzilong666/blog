异常处理 ​

官方文档

程序运行时可能会出现各种错误，如文件不存在、网络超时、类型错误等。Python 的异常处理机制允许我们优雅地处理这些错误，避免程序崩溃。本文将介绍 Python 的异常处理方法。

什么是异常？ ​

异常是程序运行时发生的错误事件，会中断正常的执行流程。

python
# 常见异常示例
# print(1 / 0)           # ZeroDivisionError
# print(undefined_var)   # NameError
# int("abc")             # ValueError
# [1, 2][10]             # IndexError
# {}["key"]              # KeyError

try-except 语句 ​

基本语法 ​

python
try:
    # 可能抛出异常的代码
    result = 10 / 0
except ZeroDivisionError:
    # 处理特定异常
    print("除数不能为零")

捕获多个异常 ​

python
try:
    num = int(input("请输入数字: "))
    result = 10 / num
except ZeroDivisionError:
    print("除数不能为零")
except ValueError:
    print("请输入有效的数字")

捕获所有异常 ​

python
try:
    # 危险操作
    result = dangerous_operation()
except Exception as e:
    print(f"发生错误: {e}")

⚠️ 注意：避免裸 except，它会捕获包括 KeyboardInterrupt 在内的所有异常。

else 和 finally ​

else 子句 ​

没有异常时执行。

python
try:
    result = 10 / 2
except ZeroDivisionError:
    print("除数不能为零")
else:
    print(f"结果是: {result}")  # 结果是: 5.0

finally 子句 ​

无论是否异常都执行。

python
try:
    f = open("file.txt", "r")
    content = f.read()
except FileNotFoundError:
    print("文件不存在")
finally:
    f.close()  # 总是执行
    print("清理完成")

完整结构 ​

python
try:
    # 尝试执行的代码
    result = 10 / 2
except ZeroDivisionError:
    # 处理异常
    print("错误")
else:
    # 无异常时执行
    print(f"成功: {result}")
finally:
    # 总是执行
    print("结束")

常见异常类型 ​

python
# AttributeError - 属性不存在
obj = None
# obj.method()  # AttributeError

# ImportError - 导入失败
# import nonexistent_module  # ImportError

# IndexError - 索引越界
# [1, 2, 3][10]  # IndexError

# KeyError - 键不存在
# {}["missing"]  # KeyError

# TypeError - 类型错误
# "2" + 2  # TypeError

# ValueError - 值错误
# int("abc")  # ValueError

# FileNotFoundError - 文件不存在
# open("nonexistent.txt")  # FileNotFoundError

获取异常信息 ​

python
try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"异常类型: {type(e).__name__}")  # ZeroDivisionError
    print(f"异常信息: {e}")                  # division by zero
    import traceback
    traceback.print_exc()  # 打印完整堆栈跟踪

主动抛出异常 ​

使用 raise 主动抛出异常。

python
def set_age(age):
    if not isinstance(age, int):
        raise TypeError("年龄必须是整数")
    if age < 0 or age > 150:
        raise ValueError("年龄必须在 0-150 之间")
    return age

try:
    set_age(-1)
except ValueError as e:
    print(f"错误: {e}")  # 错误: 年龄必须在 0-150 之间

重新抛出异常 ​

python
try:
    result = 10 / 0
except ZeroDivisionError:
    print("记录日志")
    raise  # 重新抛出原异常

自定义异常 ​

创建自己的异常类。

python
class AppError(Exception):
    """应用基础异常"""
    pass

class ValidationError(AppError):
    """验证错误"""
    def __init__(self, field, message):
        self.field = field
        self.message = message
        super().__init__(f"{field}: {message}")

class NotFoundError(AppError):
    """资源未找到"""
    pass

# 使用
def validate_user(data):
    if not data.get("name"):
        raise ValidationError("name", "姓名不能为空")
    if not data.get("email"):
        raise ValidationError("email", "邮箱不能为空")

try:
    validate_user({"name": "张三"})
except ValidationError as e:
    print(f"验证失败 - {e.field}: {e.message}")

异常链 ​

Python 3 支持异常链，保留原始异常信息。

python
try:
    int("abc")
except ValueError as e:
    raise RuntimeError("处理失败") from e

# 输出:
# RuntimeError: 处理失败
# The above exception was the direct cause of the following exception:
# ValueError: invalid literal for int() with base 10: 'abc'

抑制异常链 ​

python
try:
    int("abc")
except ValueError:
    raise RuntimeError("处理失败") from None
# 只显示 RuntimeError，不显示原始异常

上下文管理器与异常 ​

with 语句自动处理资源清理。

python
# 文件操作
try:
    with open("file.txt", "r") as f:
        content = f.read()
except FileNotFoundError:
    print("文件不存在")

# 数据库连接
import sqlite3

try:
    conn = sqlite3.connect("database.db")
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM users")
    results = cursor.fetchall()
except sqlite3.Error as e:
    print(f"数据库错误: {e}")
finally:
    conn.close()

最佳实践 ​

1. 精确捕获异常 ​

python
# 不好
try:
    do_something()
except:
    pass

# 好
try:
    do_something()
except SpecificError as e:
    handle_error(e)

2. 不要静默异常 ​

python
# 不好
try:
    risky_operation()
except Exception:
    pass  # 吞掉异常

# 好
try:
    risky_operation()
except Exception as e:
    logger.error(f"操作失败: {e}")
    raise

3. 使用异常层次结构 ​

python
class BaseAPIError(Exception):
    pass

class AuthenticationError(BaseAPIError):
    pass

class RateLimitError(BaseAPIError):
    pass

# 可以捕获所有 API 错误
try:
    call_api()
except BaseAPIError as e:
    handle_api_error(e)

4. 清理资源 ​

python
# 使用 finally 或 with
resource = acquire_resource()
try:
    use_resource(resource)
finally:
    release_resource(resource)

# 或使用上下文管理器
with managed_resource() as resource:
    use_resource(resource)

实际应用示例 ​

1. 重试机制 ​

python
import time

def retry(func, max_attempts=3, delay=1):
    """重试装饰器"""
    for attempt in range(1, max_attempts + 1):
        try:
            return func()
        except Exception as e:
            if attempt == max_attempts:
                raise
            print(f"尝试 {attempt} 失败: {e}，{delay}秒后重试")
            time.sleep(delay)

# 使用
def unstable_operation():
    import random
    if random.random() < 0.7:
        raise ConnectionError("连接失败")
    return "成功"

result = retry(unstable_operation)
print(result)

2. 输入验证 ​

python
def get_positive_int(prompt):
    while True:
        try:
            value = int(input(prompt))
            if value <= 0:
                raise ValueError("必须为正整数")
            return value
        except ValueError as e:
            print(f"无效输入: {e}")

age = get_positive_int("请输入年龄: ")

3. 安全的 JSON 解析 ​

python
import json

def safe_json_loads(json_str, default=None):
    try:
        return json.loads(json_str)
    except (json.JSONDecodeError, TypeError):
        return default

data = safe_json_loads('{"name": "张三"}')
print(data)  # {'name': '张三'}

invalid = safe_json_loads("invalid", default={})
print(invalid)  # {}

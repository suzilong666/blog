# 智能客服系统实战

## 项目概述

构建一个基于 RAG 的智能客服系统，能够自动回答用户关于产品、服务、政策等方面的问题。

### 功能特性

- ✅ 知识库管理（PDF、Word、TXT）
- ✅ 智能问答（基于 RAG）
- ✅ 多轮对话（上下文理解）
- ✅ 人工接管（无法回答时转人工）
- ✅ 对话记录和分析

### 技术栈

- **后端**: FastAPI + Python
- **AI**: OpenAI GPT-3.5-turbo + LangChain
- **向量数据库**: Chroma
- **前端**: 简单的 HTML/JavaScript

### 2. 知识库管理

`app/knowledge_base.py`:
```python
"""
知识库管理模块
负责文档加载、分割、向量化和存储
"""
from pathlib import Path
from langchain.document_loaders import (
    PyPDFLoader,
    Docx2txtLoader,
    TextLoader
)
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import Chroma
from typing import List

class KnowledgeBase:
    """知识库管理器"""
    
    def __init__(self, docs_dir: str = "./data/documents", 
                 db_dir: str = "./data/chroma_db"):
        self.docs_dir = Path(docs_dir)
        self.db_dir = db_dir
        self.embeddings = OpenAIEmbeddings()
        self.vectorstore = None
        
        # 确保目录存在
        self.docs_dir.mkdir(parents=True, exist_ok=True)
        
        # 初始化或加载向量数据库
        self._init_vectorstore()
    
    def _init_vectorstore(self):
        """初始化向量数据库"""
        try:
            # 尝试加载已存在的数据库
            self.vectorstore = Chroma(
                persist_directory=self.db_dir,
                embedding_function=self.embeddings
            )
            print(f"✅ 加载已有知识库")
        except:
            # 创建新的数据库
            self.vectorstore = Chroma(
                persist_directory=self.db_dir,
                embedding_function=self.embeddings
            )
            print("🆕 创建新知识库")
    
    def add_document(self, file_path: str) -> dict:
        """添加文档到知识库"""
        path = Path(file_path)
        
        if not path.exists():
            return {"success": False, "error": "文件不存在"}
        
        try:
            # 根据文件类型选择加载器
            loader = self._get_loader(path)
            documents = loader.load()
            
            # 分割文本
            text_splitter = RecursiveCharacterTextSplitter(
                chunk_size=1000,
                chunk_overlap=200
            )
            chunks = text_splitter.split_documents(documents)
            
            # 添加到向量数据库
            self.vectorstore.add_documents(chunks)
            self.vectorstore.persist()
            
            return {
                "success": True,
                "message": f"成功添加 {len(chunks)} 个片段",
                "filename": path.name
            }
        
        except Exception as e:
            return {"success": False, "error": str(e)}
    
    def _get_loader(self, file_path: Path):
        """根据文件类型获取加载器"""
        suffix = file_path.suffix.lower()
        
        if suffix == '.pdf':
            return PyPDFLoader(str(file_path))
        elif suffix == '.docx':
            return Docx2txtLoader(str(file_path))
        elif suffix == '.txt':
            return TextLoader(str(file_path))
        else:
            raise ValueError(f"不支持的文件类型: {suffix}")
    
    def search(self, query: str, k: int = 3) -> List[str]:
        """搜索相关文档"""
        results = self.vectorstore.similarity_search(query, k=k)
        return [doc.page_content for doc in results]
    
    def clear(self):
        """清空知识库"""
        self.vectorstore.delete_collection()
        self.vectorstore = Chroma(
            persist_directory=self.db_dir,
            embedding_function=self.embeddings
        )
        self.vectorstore.persist()
        return {"success": True, "message": "知识库已清空"}
```

### 3. 聊天引擎

`app/chat.py`:
```python
"""
聊天引擎模块
处理用户对话，结合 RAG 生成回答
"""
from langchain.chat_models import ChatOpenAI
from langchain.chains import ConversationalRetrievalChain
from langchain.memory import ConversationBufferMemory
from langchain.prompts import PromptTemplate
from .knowledge_base import KnowledgeBase

class ChatEngine:
    """聊天引擎"""
    
    def __init__(self, knowledge_base: KnowledgeBase):
        self.kb = knowledge_base
        self.llm = ChatOpenAI(
            model="gpt-3.5-turbo",
            temperature=0.3
        )
        
        # 自定义 Prompt
        prompt_template = """你是智能客服助手。基于以下上下文回答问题。

规则：
1. 只基于提供的信息回答
2. 如果上下文中没有相关信息，回复"抱歉，我暂时无法回答这个问题，将为您转接人工客服"
3. 保持友好、专业的语气
4. 回答简洁明了

上下文：
{context}

对话历史：
{chat_history}

用户问题：{question}

回答："""

        PROMPT = PromptTemplate(
            template=prompt_template,
            input_variables=["context", "chat_history", "question"]
        )
        
        # 创建对话链
        self.conversation_chain = ConversationalRetrievalChain.from_llm(
            llm=self.llm,
            retriever=knowledge_base.vectorstore.as_retriever(
                search_kwargs={"k": 3}
            ),
            memory=ConversationBufferMemory(
                memory_key="chat_history",
                return_messages=True
            ),
            combine_docs_chain_kwargs={"prompt": PROMPT},
            return_source_documents=True
        )
    
    def chat(self, user_message: str) -> dict:
        """处理用户消息"""
        try:
            result = self.conversation_chain.invoke({
                "question": user_message
            })
            
            answer = result["answer"]
            sources = result.get("source_documents", [])
            
            # 判断是否需要转人工
            needs_human = "转接人工客服" in answer
            
            return {
                "success": True,
                "answer": answer,
                "needs_human": needs_human,
                "sources": [
                    doc.page_content[:200] 
                    for doc in sources[:2]
                ] if sources else []
            }
        
        except Exception as e:
            return {
                "success": False,
                "error": str(e),
                "answer": "抱歉，系统出现错误，请稍后重试",
                "needs_human": True
            }
```

### 4. FastAPI 主应用

`app/main.py`:
```python
"""
FastAPI 主应用
提供 RESTful API 接口
"""
from fastapi import FastAPI, UploadFile, File, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from typing import List, Optional
import shutil
from pathlib import Path

from .knowledge_base import KnowledgeBase
from .chat import ChatEngine

# 初始化 FastAPI
app = FastAPI(
    title="智能客服系统",
    description="基于 RAG 的智能客服 API",
    version="1.0.0"
)

# 配置 CORS
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"]
)

# 初始化组件
kb = KnowledgeBase()
chat_engine = ChatEngine(kb)

# 数据模型
class ChatRequest(BaseModel):
    message: str
    session_id: Optional[str] = None

class ChatResponse(BaseModel):
    success: bool
    answer: str
    needs_human: bool
    sources: List[str] = []
    error: Optional[str] = None

class UploadResponse(BaseModel):
    success: bool
    message: str
    filename: Optional[str] = None

# API 路由

@app.get("/")
def root():
    """根路径"""
    return {
        "message": "智能客服系统 API",
        "docs": "/docs"
    }

@app.post("/api/chat", response_model=ChatResponse)
def chat(request: ChatRequest):
    """聊天接口"""
    result = chat_engine.chat(request.message)
    
    if not result["success"]:
        raise HTTPException(status_code=500, detail=result.get("error"))
    
    return ChatResponse(**result)

@app.post("/api/upload", response_model=UploadResponse)
async def upload_document(file: UploadFile = File(...)):
    """上传文档到知识库"""
    # 保存文件
    file_path = Path(f"./data/documents/{file.filename}")
    
    with open(file_path, "wb") as buffer:
        shutil.copyfileobj(file.file, buffer)
    
    # 添加到知识库
    result = kb.add_document(str(file_path))
    
    if not result["success"]:
        raise HTTPException(status_code=500, detail=result.get("error"))
    
    return UploadResponse(
        success=True,
        message=result["message"],
        filename=result["filename"]
    )

@app.post("/api/knowledge/clear")
def clear_knowledge():
    """清空知识库"""
    result = kb.clear()
    return result

@app.get("/api/health")
def health_check():
    """健康检查"""
    return {"status": "healthy"}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

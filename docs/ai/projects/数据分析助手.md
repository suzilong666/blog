# 数据分析助手实战

## 项目概述

构建一个自然语言驱动的数据分析助手，用户可以通过对话方式查询和分析数据。

### 功能特性

- ✅ 自然语言查询（"显示上季度销售额"）
- ✅ 自动 SQL 生成
- ✅ 数据可视化
- ✅ 洞察报告生成
- ✅ 多数据源支持

## 技术架构

```
用户提问 → NLP理解 → SQL生成 → 数据库查询 → 结果分析 → 可视化 → 回答
```

## 实现方案

### 1. 核心组件

```python
"""
数据分析助手核心引擎
"""
import pandas as pd
from openai import OpenAI
from langchain.chat_models import ChatOpenAI
from langchain.utilities import SQLDatabase

class DataAnalyst:
    """智能数据分析助手"""
    
    def __init__(self, db_uri: str = "sqlite:///data.db"):
        self.client = OpenAI()
        self.llm = ChatOpenAI(model="gpt-4", temperature=0)
        self.db = SQLDatabase.from_uri(db_uri)
    
    def query_to_sql(self, question: str) -> str:
        """将自然语言转换为 SQL"""
        
        # 获取数据库 schema
        schema = self.db.get_table_info()
        
        prompt = f"""你是一个 SQL 专家。根据以下数据库结构和用户问题，生成对应的 SQL 查询。

数据库结构：
{schema}

用户问题：{question}

要求：
1. 只输出 SQL 语句
2. 使用标准 SQL 语法
3. 确保查询高效
4. 添加必要的注释

SQL："""

        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            temperature=0
        )
        
        sql = response.choices[0].message.content.strip()
        return sql
    
    def execute_query(self, sql: str) -> pd.DataFrame:
        """执行 SQL 查询"""
        try:
            result = self.db.run(sql)
            # 解析结果为 DataFrame
            # 实际项目中需要根据返回格式解析
            return pd.DataFrame(result)
        except Exception as e:
            raise Exception(f"查询执行失败: {str(e)}")
    
    def analyze_results(self, question: str, data: pd.DataFrame) -> dict:
        """分析查询结果"""
        
        # 生成数据统计
        stats = {
            "rows": len(data),
            "columns": list(data.columns),
            "summary": data.describe().to_dict() if len(data) > 0 else {}
        }
        
        # AI 分析洞察
        prompt = f"""基于以下数据和原始问题，提供分析洞察：

原始问题：{question}

数据统计：
{stats}

数据样本（前5行）：
{data.head().to_string()}

请提供：
1. 关键发现
2. 趋势分析
3. 异常点
4. 业务建议

用简洁的语言，分点说明。"""

        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}]
        )
        
        insights = response.choices[0].message.content
        
        return {
            "data": data,
            "stats": stats,
            "insights": insights
        }
    
    def generate_chart_code(self, data: pd.DataFrame, chart_type: str = "auto") -> str:
        """生成可视化代码"""
        
        # 自动选择图表类型
        if chart_type == "auto":
            if len(data) <= 10:
                chart_type = "bar"
            elif len(data) > 100:
                chart_type = "line"
            else:
                chart_type = "scatter"
        
        prompt = f"""为以下数据生成 Python 可视化代码（使用 matplotlib 或 seaborn）：

数据列：{list(data.columns)}
数据类型：{chart_type}
数据样本：
{data.head().to_string()}

要求：
1. 完整的可运行代码
2. 包含中文支持
3. 添加标题和标签
4. 美观的配色

只输出代码，不要解释。"""

        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}]
        )
        
        return response.choices[0].message.content
    
    def analyze(self, question: str) -> dict:
        """完整分析流程"""
        
        # Step 1: 生成 SQL
        sql = self.query_to_sql(question)
        
        # Step 2: 执行查询
        data = self.execute_query(sql)
        
        # Step 3: 分析结果
        analysis = self.analyze_results(question, data)
        
        # Step 4: 生成可视化代码
        chart_code = self.generate_chart_code(data)
        
        return {
            "sql": sql,
            "data": analysis["data"],
            "insights": analysis["insights"],
            "chart_code": chart_code
        }
```

### 2. FastAPI 服务

```python
"""
数据分析 API 服务
"""
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import base64
import io

app = FastAPI(title="数据分析助手")
analyst = DataAnalyst("sqlite:///sales.db")

class QueryRequest(BaseModel):
    question: str

@app.post("/api/analyze")
def analyze(request: QueryRequest):
    """数据分析接口"""
    try:
        result = analyst.analyze(request.question)
        
        # 将 DataFrame 转换为 JSON
        data_json = result["data"].to_dict(orient="records")
        
        return {
            "success": True,
            "sql": result["sql"],
            "data": data_json,
            "insights": result["insights"],
            "chart_code": result["chart_code"]
        }
    
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/api/tables")
def list_tables():
    """列出所有表"""
    tables = analyst.db.get_usable_table_names()
    return {"tables": tables}

@app.get("/api/schema/{table_name}")
def get_schema(table_name: str):
    """获取表结构"""
    schema = analyst.db.get_table_info([table_name])
    return {"schema": schema}
```

## 使用示例

### 示例 1：销售分析

**用户提问：**
```
显示上个月销售额最高的5个产品
```

**系统执行：**
1. 生成 SQL：
```sql
SELECT product_name, SUM(amount) as total_sales
FROM orders
WHERE order_date >= DATE('now', '-1 month')
GROUP BY product_name
ORDER BY total_sales DESC
LIMIT 5;
```

2. 返回结果并生成洞察：
```
关键发现：
1. iPhone 15 Pro 以 250万销售额位居榜首
2. Top 5 产品占总销售额的 45%
3. 电子产品占据主导地位

建议：
- 增加热门产品的库存
- 考虑捆绑销售策略
```

# 多模态应用开发

## 什么是多模态？

多模态（Multimodal）指的是模型能够同时处理和理解多种类型的信息，包括：
- **文本**（Text）
- **图像**（Image）
- **音频**（Audio）
- **视频**（Video）

传统的语言模型只能处理文本，而多模态模型可以"看"、"听"、"读"，实现更接近人类的感知能力。

## 主流多模态模型

| 模型 | 开发商 | 支持的模态 | 特点 |
|------|--------|-----------|------|
| GPT-4 Vision | OpenAI | 文本 + 图像 | 强大的图像理解 |
| GPT-4o | OpenAI | 文本 + 图像 + 音频 | 原生多模态，速度快 |
| Claude 3 | Anthropic | 文本 + 图像 | 优秀的视觉推理 |
| Gemini Pro Vision | Google | 文本 + 图像 + 视频 | Google 生态集成 |
| Qwen-VL | 阿里 | 文本 + 图像 | 中文优化 |

## GPT-4 Vision API

### 基础用法

```python
from openai import OpenAI
import base64

client = OpenAI()

def analyze_image(image_path: str, prompt: str = "描述这张图片") -> str:
    """分析图片内容"""
    
    # 将图片编码为 base64
    with open(image_path, "rb") as image_file:
        base64_image = base64.b64encode(image_file.read()).decode('utf-8')
    
    response = client.chat.completions.create(
        model="gpt-4-vision-preview",
        messages=[
            {
                "role": "user",
                "content": [
                    {"type": "text", "text": prompt},
                    {
                        "type": "image_url",
                        "image_url": {
                            "url": f"data:image/jpeg;base64,{base64_image}"
                        }
                    }
                ]
            }
        ],
        max_tokens=500
    )
    
    return response.choices[0].message.content

# 使用
description = analyze_image("photo.jpg", "这张图片里有什么？")
print(description)
```

## Claude 3 视觉能力

Claude 3 同样支持图像理解，API 调用方式类似：

```python
from anthropic import Anthropic
import base64

client = Anthropic()

def claude_vision(image_path: str, prompt: str) -> str:
    """使用 Claude 3 分析图片"""
    
    with open(image_path, "rb") as f:
        base64_image = base64.b64encode(f.read()).decode('utf-8')
    
    # 确定媒体类型
    media_type = "image/jpeg" if image_path.endswith('.jpg') else "image/png"
    
    message = client.messages.create(
        model="claude-3-opus-20240229",
        max_tokens=1000,
        messages=[
            {
                "role": "user",
                "content": [
                    {
                        "type": "image",
                        "source": {
                            "type": "base64",
                            "media_type": media_type,
                            "data": base64_image
                        }
                    },
                    {
                        "type": "text",
                        "text": prompt
                    }
                ]
            }
        ]
    )
    
    return message.content[0].text
```

## 视频处理

### 提取帧进行分析

```python
import cv2
from openai import OpenAI
import base64

def analyze_video(video_path: str, num_frames: int = 5) -> str:
    """分析视频内容"""
    
    # 打开视频
    cap = cv2.VideoCapture(video_path)
    total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
    
    # 均匀采样帧
    frame_indices = [int(i * total_frames / num_frames) for i in range(num_frames)]
    frames = []
    
    for idx in frame_indices:
        cap.set(cv2.CAP_PROP_POS_FRAMES, idx)
        ret, frame = cap.read()
        if ret:
            # 转换为 JPEG
            _, buffer = cv2.imencode('.jpg', frame)
            base64_frame = base64.b64encode(buffer).decode('utf-8')
            frames.append(base64_frame)
    
    cap.release()
    
    # 构建消息
    content = [{"type": "text", "text": "分析这个视频的内容"}]
    
    for frame in frames:
        content.append({
            "type": "image_url",
            "image_url": f"data:image/jpeg;base64,{frame}"
        })
    
    # 调用 GPT-4 Vision
    client = OpenAI()
    response = client.chat.completions.create(
        model="gpt-4-vision-preview",
        messages=[{"role": "user", "content": content}],
        max_tokens=1000
    )
    
    return response.choices[0].message.content
```

## 实战项目：智能相册管理器

```python
import os
import json
from pathlib import Path
from openai import OpenAI
import base64
from datetime import datetime

class SmartPhotoManager:
    """智能相册管理器"""
    
    def __init__(self, photo_dir: str):
        self.photo_dir = Path(photo_dir)
        self.client = OpenAI()
        self.metadata_file = self.photo_dir / "metadata.json"
        self.metadata = self._load_metadata()
    
    def _load_metadata(self) -> dict:
        """加载元数据"""
        if self.metadata_file.exists():
            with open(self.metadata_file, 'r') as f:
                return json.load(f)
        return {}
    
    def _save_metadata(self):
        """保存元数据"""
        with open(self.metadata_file, 'w') as f:
            json.dump(self.metadata, f, indent=2, ensure_ascii=False)
    
    def analyze_photo(self, image_path: str) -> dict:
        """分析单张照片"""
        
        with open(image_path, "rb") as f:
            base64_image = base64.b64encode(f.read()).decode('utf-8')
        
        response = self.client.chat.completions.create(
            model="gpt-4-vision-preview",
            messages=[
                {
                    "role": "user",
                    "content": [
                        {"type": "text", "text": """
                        分析这张照片，提供以下信息（JSON格式）：
                        {
                            "scene": "场景类型（如：海滩、山脉、城市）",
                            "objects": ["主要物体列表"],
                            "people_count": "人物数量",
                            "mood": "氛围（如：欢乐、宁静、浪漫）",
                            "colors": ["主色调"],
                            "time_of_day": "拍摄时间（如：早晨、黄昏）",
                            "tags": ["标签1", "标签2"]
                        }
                        """},
                        {
                            "type": "image_url",
                            "image_url": f"data:image/jpeg;base64,{base64_image}"
                        }
                    ]
                }
            ],
            max_tokens=500
        )
        
        try:
            return json.loads(response.choices[0].message.content)
        except:
            return {}
    
    def process_all_photos(self):
        """处理所有照片"""
        image_extensions = {'.jpg', '.jpeg', '.png', '.gif', '.webp'}
        
        for image_path in self.photo_dir.iterdir():
            if image_path.suffix.lower() in image_extensions:
                file_name = image_path.name
                
                if file_name not in self.metadata:
                    print(f"分析: {file_name}")
                    analysis = self.analyze_photo(str(image_path))
                    
                    self.metadata[file_name] = {
                        "analysis": analysis,
                        "processed_at": datetime.now().isoformat()
                    }
                    
                    self._save_metadata()
    
    def search_photos(self, query: str) -> list:
        """搜索照片"""
        results = []
        
        for filename, data in self.metadata.items():
            analysis = data.get("analysis", {})
            
            # 在场景、物体、标签中搜索
            searchable_text = " ".join([
                str(analysis.get("scene", "")),
                " ".join(analysis.get("objects", [])),
                " ".join(analysis.get("tags", [])),
                analysis.get("mood", "")
            ]).lower()
            
            if query.lower() in searchable_text:
                results.append({
                    "filename": filename,
                    "analysis": analysis
                })
        
        return results
    
    def generate_album(self, theme: str) -> list:
        """根据主题生成相册"""
        all_photos = []
        
        for filename, data in self.metadata.items():
            mood = data.get("analysis", {}).get("mood", "").lower()
            scene = data.get("analysis", {}).get("scene", "").lower()
            
            if theme.lower() in mood or theme.lower() in scene:
                all_photos.append(filename)
        
        return all_photos

# 使用
manager = SmartPhotoManager("./photos")

# 处理所有照片
manager.process_all_photos()

# 搜索
results = manager.search_photos("海滩")
for photo in results:
    print(f"{photo['filename']}: {photo['analysis']}")

# 生成主题相册
sunset_album = manager.generate_album("黄昏")
```

## 性能优化

### 1. 图片压缩

```python
from PIL import Image
import io

def compress_image(image_path: str, max_size: int = 1024) -> bytes:
    """压缩图片以减少 token 消耗"""
    
    img = Image.open(image_path)
    
    # 调整大小
    if max(img.size) > max_size:
        ratio = max_size / max(img.size)
        new_size = tuple(int(dim * ratio) for dim in img.size)
        img = img.resize(new_size, Image.Resampling.LANCZOS)
    
    # 保存到内存
    buffer = io.BytesIO()
    img.save(buffer, format='JPEG', quality=85)
    
    return buffer.getvalue()
```

### 2. 批量处理

```python
import asyncio
from concurrent.futures import ThreadPoolExecutor

async def batch_analyze(images: list, max_workers: int = 5):
    """批量分析图片"""
    
    def analyze_single(image_path):
        return analyze_image(image_path)
    
    loop = asyncio.get_event_loop()
    
    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = [
            loop.run_in_executor(executor, analyze_single, img)
            for img in images
        ]
        
        results = await asyncio.gather(*futures)
    
    return results
```

### 3. 缓存结果

```python
import hashlib

def get_image_hash(image_path: str) -> str:
    """计算图片哈希值"""
    with open(image_path, 'rb') as f:
        return hashlib.md5(f.read()).hexdigest()

# 使用哈希作为缓存键
cache = {}

def cached_analyze(image_path: str):
    img_hash = get_image_hash(image_path)
    
    if img_hash in cache:
        return cache[img_hash]
    
    result = analyze_image(image_path)
    cache[img_hash] = result
    
    return result
```

## 成本考虑

### GPT-4 Vision 定价

- **输入**：$0.01 / 1K tokens（包含图片）
- **输出**：$0.03 / 1K tokens

**图片 token 计算：**
- 低分辨率模式：固定 85 tokens
- 高分辨率模式：根据图片大小计算

### 优化建议

1. **选择合适的分辨率**
   ```python
   # 低分辨率（更便宜）
   image_url = {
       "url": "...",
       "detail": "low"  # 默认是 "high"
   }
   ```

2. **预处理图片**
   - 裁剪无关区域
   - 压缩文件大小
   - 调整到合适尺寸

3. **缓存分析结果**
   - 避免重复分析相同图片

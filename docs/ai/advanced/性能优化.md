## Token 计费机制

### 什么是 Token？

Token 是大语言模型处理文本的基本单位。一个 token 可以是：
- 一个完整的单词（如 "hello"）
- 一个词的一部分（如 "ing"）
- 一个标点符号

**经验法则：**
- 英文：1 token ≈ 4 个字符 ≈ 0.75 个单词
- 中文：1 token ≈ 1-2 个汉字

### 计费方式

```
总成本 = (输入 tokens × 输入单价) + (输出 tokens × 输出单价)
```

**OpenAI 定价示例（GPT-3.5-turbo）：**
- 输入：$0.0005 / 1K tokens
- 输出：$0.0015 / 1K tokens

**示例计算：**
```
输入 500 tokens + 输出 200 tokens
成本 = (500/1000 × 0.0005) + (200/1000 × 0.0015)
     = 0.00025 + 0.0003
     = $0.00055
```

### 计算对话的 Token

```python
def count_message_tokens(messages: list, model: str = "gpt-3.5-turbo") -> int:
    """计算对话消息的 token 数量"""
    encoding = tiktoken.encoding_for_model(model)
    
    # 每个消息的基础开销
    tokens_per_message = 4
    
    total_tokens = 0
    for message in messages:
        total_tokens += tokens_per_message
        total_tokens += len(encoding.encode(message.get("role", "")))
        total_tokens += len(encoding.encode(message.get("content", "")))
    
    # 每条回复的额外开销
    total_tokens += 2
    
    return total_tokens

# 使用
messages = [
    {"role": "system", "content": "你是一个助手"},
    {"role": "user", "content": "你好"},
    {"role": "assistant", "content": "你好！有什么可以帮助你的？"}
]

tokens = count_message_tokens(messages)
print(f"对话总 token: {tokens}")
```

## 成本估算工具

```python
class CostCalculator:
    """API 成本计算器"""
    
    PRICING = {
        "gpt-3.5-turbo": {"input": 0.0005, "output": 0.0015},
        "gpt-4": {"input": 0.03, "output": 0.06},
        "gpt-4-turbo": {"input": 0.01, "output": 0.03},
        "claude-3-sonnet": {"input": 0.003, "output": 0.015},
        "claude-3-haiku": {"input": 0.00025, "output": 0.00125}
    }
    
    @staticmethod
    def estimate(input_tokens: int, output_tokens: int, model: str = "gpt-3.5-turbo") -> float:
        """估算成本（美元）"""
        prices = CostCalculator.PRICING.get(model, CostCalculator.PRICING["gpt-3.5-turbo"])
        
        input_cost = (input_tokens / 1000) * prices["input"]
        output_cost = (output_tokens / 1000) * prices["output"]
        
        return round(input_cost + output_cost, 6)
    
    @staticmethod
    def estimate_from_text(input_text: str, estimated_output_tokens: int, model: str = "gpt-3.5-turbo") -> dict:
        """从文本估算成本"""
        input_tokens = count_tokens(input_text, model)
        cost = CostCalculator.estimate(input_tokens, estimated_output_tokens, model)
        
        return {
            "input_tokens": input_tokens,
            "output_tokens": estimated_output_tokens,
            "total_tokens": input_tokens + estimated_output_tokens,
            "cost_usd": cost,
            "cost_cny": cost * 7.2  # 转换为人民币
        }

# 使用
result = CostCalculator.estimate_from_text(
    "请帮我分析这段代码...",
    estimated_output_tokens=500,
    model="gpt-3.5-turbo"
)

print(f"输入 tokens: {result['input_tokens']}")
print(f"预计输出 tokens: {result['output_tokens']}")
print(f"预计成本: ${result['cost_usd']} (¥{result['cost_cny']:.4f})")
```

### 4. 缓存策略

#### 内存缓存

```python
import hashlib
import json
from functools import lru_cache

class ResponseCache:
    """API 响应缓存"""
    
    def __init__(self):
        self.cache = {}
    
    def _generate_key(self, messages: list, model: str) -> str:
        """生成缓存键"""
        content = json.dumps({"messages": messages, "model": model})
        return hashlib.md5(content.encode()).hexdigest()
    
    def get(self, messages: list, model: str) -> str:
        """获取缓存"""
        key = self._generate_key(messages, model)
        return self.cache.get(key)
    
    def set(self, messages: list, model: str, response: str):
        """设置缓存"""
        key = self._generate_key(messages, model)
        self.cache[key] = response
    
    def clear(self):
        """清空缓存"""
        self.cache.clear()

# 使用
cache = ResponseCache()

def cached_chat(messages: list, model: str = "gpt-3.5-turbo") -> str:
    """带缓存的聊天"""
    # 检查缓存
    cached_response = cache.get(messages, model)
    if cached_response:
        print("命中缓存")
        return cached_response
    
    # 调用 API
    response = client.chat.completions.create(
        model=model,
        messages=messages
    )
    
    result = response.choices[0].message.content
    
    # 存入缓存
    cache.set(messages, model, result)
    
    return result
```

#### Redis 缓存（生产环境）

```python
import redis
import json

class RedisCache:
    """Redis 缓存"""
    
    def __init__(self, host='localhost', port=6379, db=0):
        self.redis_client = redis.Redis(host=host, port=port, db=db)
        self.ttl = 3600  # 缓存1小时
    
    def get(self, key: str) -> str:
        """获取缓存"""
        data = self.redis_client.get(key)
        if data:
            return json.loads(data)
        return None
    
    def set(self, key: str, value: str):
        """设置缓存"""
        self.redis_client.setex(key, self.ttl, json.dumps(value))

# 使用
redis_cache = RedisCache()
```

### 自定义追踪器

```python
import time
from dataclasses import dataclass, field
from typing import List

@dataclass
class APIUsageRecord:
    """API 使用记录"""
    timestamp: float = field(default_factory=time.time)
    model: str = ""
    input_tokens: int = 0
    output_tokens: int = 0
    cost: float = 0.0
    endpoint: str = ""

class UsageTracker:
    """用量追踪器"""
    
    def __init__(self):
        self.records: List[APIUsageRecord] = []
    
    def add_record(self, record: APIUsageRecord):
        """添加记录"""
        self.records.append(record)
    
    def get_total_cost(self, period_hours: int = 24) -> float:
        """获取指定时间段内的总成本"""
        cutoff_time = time.time() - (period_hours * 3600)
        
        return sum(
            r.cost for r in self.records
            if r.timestamp > cutoff_time
        )
    
    def get_total_tokens(self, period_hours: int = 24) -> dict:
        """获取指定时间段内的 token 用量"""
        cutoff_time = time.time() - (period_hours * 3600)
        
        recent_records = [r for r in self.records if r.timestamp > cutoff_time]
        
        return {
            "input_tokens": sum(r.input_tokens for r in recent_records),
            "output_tokens": sum(r.output_tokens for r in recent_records),
            "total_tokens": sum(r.input_tokens + r.output_tokens for r in recent_records)
        }
    
    def generate_report(self) -> str:
        """生成用量报告"""
        total_cost = self.get_total_cost(24)
        tokens = self.get_total_tokens(24)
        
        report = f"""
=== API 用量报告（过去24小时）===
总成本: ${total_cost:.4f}
总 Token: {tokens['total_tokens']}
  - 输入: {tokens['input_tokens']}
  - 输出: {tokens['output_tokens']}
请求次数: {len(self.records)}
平均每次成本: ${total_cost / max(len(self.records), 1):.6f}
"""
        return report

# 使用
tracker = UsageTracker()

# 在每次 API 调用后记录
record = APIUsageRecord(
    model="gpt-3.5-turbo",
    input_tokens=100,
    output_tokens=200,
    cost=0.00035,
    endpoint="/chat/completions"
)
tracker.add_record(record)

# 查看报告
print(tracker.generate_report())
```

## 预算控制

### 设置预算上限

```python
class BudgetManager:
    """预算管理"""
    
    def __init__(self, daily_budget: float = 10.0):
        self.daily_budget = daily_budget
        self.tracker = UsageTracker()
    
    def can_make_request(self, estimated_cost: float) -> bool:
        """检查是否可以发起请求"""
        current_spent = self.tracker.get_total_cost(24)
        return (current_spent + estimated_cost) <= self.daily_budget
    
    def check_and_execute(self, api_call_func, estimated_cost: float):
        """检查预算并执行"""
        if not self.can_make_request(estimated_cost):
            raise BudgetExceededError(
                f"超出每日预算 (${self.daily_budget})"
            )
        
        result = api_call_func()
        return result

class BudgetExceededError(Exception):
    """预算超限异常"""
    pass

# 使用
budget = BudgetManager(daily_budget=5.0)

try:
    result = budget.check_and_execute(
        lambda: client.chat.completions.create(...),
        estimated_cost=0.001
    )
except BudgetExceededError as e:
    print(f"预算不足: {e}")
    # 降级处理：使用更便宜的模型或返回缓存
```

## 实战：成本优化系统

```python
import time
from typing import Optional

class OptimizedChatClient:
    """优化的聊天客户端"""
    
    def __init__(self, budget_limit: float = 10.0):
        self.client = OpenAI()
        self.cache = ResponseCache()
        self.budget_manager = BudgetManager(budget_limit)
        self.tracker = UsageTracker()
    
    def chat(self, messages: list, 
             model: Optional[str] = None,
             use_cache: bool = True) -> str:
        """智能聊天，自动优化成本"""
        
        # 1. 尝试缓存
        if use_cache:
            cached = self.cache.get(messages, model or "gpt-3.5-turbo")
            if cached:
                return cached
        
        # 2. 自动选择模型
        if model is None:
            model = self._select_model(messages)
        
        # 3. 估算成本
        input_tokens = count_message_tokens(messages, model)
        estimated_output = 200  # 预估输出
        estimated_cost = CostCalculator.estimate(input_tokens, estimated_output, model)
        
        # 4. 检查预算
        if not self.budget_manager.can_make_request(estimated_cost):
            # 降级到更便宜的模型
            if model != "gpt-3.5-turbo":
                print("预算紧张，降级到 gpt-3.5-turbo")
                model = "gpt-3.5-turbo"
            else:
                raise BudgetExceededError("预算已用尽")
        
        # 5. 调用 API
        start_time = time.time()
        response = self.client.chat.completions.create(
            model=model,
            messages=messages,
            max_tokens=500
        )
        
        result = response.choices[0].message.content
        
        # 6. 记录用量
        usage = response.usage
        actual_cost = CostCalculator.estimate(
            usage.prompt_tokens,
            usage.completion_tokens,
            model
        )
        
        record = APIUsageRecord(
            model=model,
            input_tokens=usage.prompt_tokens,
            output_tokens=usage.completion_tokens,
            cost=actual_cost,
            endpoint="/chat/completions"
        )
        self.tracker.add_record(record)
        
        # 7. 缓存结果
        self.cache.set(messages, model, result)
        
        elapsed = time.time() - start_time
        print(f"耗时: {elapsed:.2f}s | 成本: ${actual_cost:.6f}")
        
        return result
    
    def _select_model(self, messages: list) -> str:
        """根据消息复杂度选择模型"""
        # 简单判断：消息长度和关键词
        total_length = sum(len(m.get("content", "")) for m in messages)
        
        if total_length < 500:
            return "gpt-3.5-turbo"
        else:
            return "gpt-4"
    
    def get_usage_report(self) -> str:
        """获取用量报告"""
        return self.tracker.generate_report()

# 使用
client = OptimizedChatClient(budget_limit=5.0)

response = client.chat([
    {"role": "user", "content": "你好"}
])

print(client.get_usage_report())
```

# 魔术方法

[官方文档](https://docs.python.org/zh-cn/3/reference/datamodel.html#special-method-names)

魔术方法（也称为特殊方法或双下划线方法）是以 `__` 开头和结尾的方法，它们允许我们自定义类的行为。通过实现这些方法，我们可以让类的实例支持内置操作，如打印、比较、算术运算等。

## __str__ 和 __repr__

这两个方法控制对象的字符串表示。

### __str__

面向用户，用于 `print()` 和 `str()`。

```python
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def __str__(self):
        return f"{self.name}, {self.age}岁"

person = Person("张三", 25)
print(person)       # 张三, 25岁
print(str(person))  # 张三, 25岁
```

### __repr__

面向开发者，用于调试，应尽可能明确。

```python
class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def __repr__(self):
        return f"Person('{self.name}', {self.age})"

person = Person("张三", 25)
print(repr(person))  # Person('张三', 25)

# 在交互式解释器中直接输入变量名会调用 __repr__
# >>> person
# Person('张三', 25)
```

> 💡 **最佳实践**：始终实现 `__repr__`，`__str__` 可选。理想情况下 `eval(repr(obj)) == obj`。

## __len__, __getitem__, __setitem__

这些方法让对象支持序列操作。

```python
class ShoppingCart:
    def __init__(self):
        self.items = []

    def add_item(self, item):
        self.items.append(item)

    def __len__(self):
        return len(self.items)

    def __getitem__(self, index):
        return self.items[index]

    def __setitem__(self, index, value):
        self.items[index] = value

    def __delitem__(self, index):
        del self.items[index]

cart = ShoppingCart()
cart.add_item("苹果")
cart.add_item("香蕉")
cart.add_item("橙子")

print(len(cart))       # 3
print(cart[0])         # 苹果
cart[1] = "葡萄"
print(cart[1])         # 葡萄
del cart[2]
print(len(cart))       # 2
```

### 支持迭代

```python
class ShoppingCart:
    def __init__(self):
        self.items = []

    def add_item(self, item):
        self.items.append(item)

    def __iter__(self):
        return iter(self.items)

cart = ShoppingCart()
cart.add_item("苹果")
cart.add_item("香蕉")

for item in cart:
    print(item)
# 苹果
# 香蕉
```

## 运算符重载

### 比较运算符

```python
class Student:
    def __init__(self, name, score):
        self.name = name
        self.score = score

    def __eq__(self, other):
        if not isinstance(other, Student):
            return NotImplemented
        return self.score == other.score

    def __lt__(self, other):
        if not isinstance(other, Student):
            return NotImplemented
        return self.score < other.score

    def __le__(self, other):
        return self == other or self < other

    def __gt__(self, other):
        return not self <= other

    def __ge__(self, other):
        return not self < other

    def __ne__(self, other):
        return not self == other

# 使用
s1 = Student("张三", 90)
s2 = Student("李四", 85)

print(s1 > s2)   # True
print(s1 == s2)  # False
print(s1 >= s2)  # True

# 支持排序
students = [s1, s2, Student("王五", 95)]
sorted_students = sorted(students)
for s in sorted_students:
    print(f"{s.name}: {s.score}")
# 李四: 85
# 张三: 90
# 王五: 95
```

### 算术运算符

```python
class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __add__(self, other):
        return Vector(self.x + other.x, self.y + other.y)

    def __sub__(self, other):
        return Vector(self.x - other.x, self.y - other.y)

    def __mul__(self, scalar):
        return Vector(self.x * scalar, self.y * scalar)

    def __eq__(self, other):
        return self.x == other.x and self.y == other.y

    def __repr__(self):
        return f"Vector({self.x}, {self.y})"

v1 = Vector(1, 2)
v2 = Vector(3, 4)

print(v1 + v2)    # Vector(4, 6)
print(v1 - v2)    # Vector(-2, -2)
print(v1 * 3)     # Vector(3, 6)
print(v1 == v2)   # False
```

## 上下文管理协议

实现 `__enter__` 和 `__exit__` 使对象支持 `with` 语句。

```python
class FileManager:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
        self.file = None

    def __enter__(self):
        self.file = open(self.filename, self.mode, encoding="utf-8")
        return self.file

    def __exit__(self, exc_type, exc_val, exc_tb):
        if self.file:
            self.file.close()
        if exc_type:
            print(f"发生异常: {exc_val}")
        return False  # 不抑制异常

# 使用
with FileManager("test.txt", "w") as f:
    f.write("Hello, World!")
# 文件自动关闭
```

## 可调用对象

实现 `__call__` 使实例可以像函数一样调用。

```python
class Adder:
    def __init__(self, n):
        self.n = n

    def __call__(self, x):
        return x + self.n

add_five = Adder(5)
print(add_five(10))  # 15
print(add_five(20))  # 25
```

### 实际应用：缓存装饰器

```python
class Memoize:
    def __init__(self, func):
        self.func = func
        self.cache = {}

    def __call__(self, *args):
        if args not in self.cache:
            self.cache[args] = self.func(*args)
        return self.cache[args]

@Memoize
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(50))  # 快速计算
```

## 属性访问控制

### __getattr__ 和 __setattr__

```python
class SmartDict:
    def __init__(self):
        self._data = {}

    def __getattr__(self, name):
        # 当属性不存在时调用
        if name in self._data:
            return self._data[name]
        raise AttributeError(f"'{type(self).__name__}' 没有属性 '{name}'")

    def __setattr__(self, name, value):
        if name.startswith("_"):
            super().__setattr__(name, value)
        else:
            self._data[name] = value

obj = SmartDict()
obj.name = "张三"
obj.age = 25
print(obj.name)  # 张三
print(obj.age)   # 25
```

## 完整示例

```python
class Matrix:
    """简单的矩阵类"""

    def __init__(self, data):
        self.data = data
        self.rows = len(data)
        self.cols = len(data[0]) if data else 0

    def __getitem__(self, index):
        return self.data[index]

    def __repr__(self):
        rows = ",\n  ".join(str(row) for row in self.data)
        return f"Matrix([\n  {rows}\n])"

    def __add__(self, other):
        if self.rows != other.rows or self.cols != other.cols:
            raise ValueError("矩阵维度不匹配")
        result = [
            [self.data[i][j] + other.data[i][j] for j in range(self.cols)]
            for i in range(self.rows)
        ]
        return Matrix(result)

    def __eq__(self, other):
        return self.data == other.data

    def transpose(self):
        result = [
            [self.data[j][i] for j in range(self.rows)]
            for i in range(self.cols)
        ]
        return Matrix(result)

# 使用
m1 = Matrix([[1, 2], [3, 4]])
m2 = Matrix([[5, 6], [7, 8]])

print(m1 + m2)
# Matrix([
#   [6, 8],
#   [10, 12]
# ])

print(m1.transpose())
# Matrix([
#   [1, 3],
#   [2, 4]
# ])
```

## 最佳实践

1. **成对实现**：如 `__eq__` 和 `__hash__`，`__lt__` 和其他比较方法
2. **返回 NotImplemented**：不支持的操作返回 `NotImplemented` 而非抛出异常
3. **保持一致性**：`__str__` 和 `__repr__` 不应矛盾
4. **不要过度使用**：只在有意义时实现魔术方法
5. **参考文档**：查阅官方文档了解所有可用的魔术方法

```python
from functools import total_ordering

@total_ordering
class Product:
    """使用装饰器自动生成比较方法"""
    def __init__(self, name, price):
        self.name = name
        self.price = price

    def __eq__(self, other):
        return self.price == other.price

    def __lt__(self, other):
        return self.price < other.price

    def __repr__(self):
        return f"Product('{self.name}', {self.price})"
```
